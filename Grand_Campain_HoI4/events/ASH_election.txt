country_event = { # Get Results
	id = ASH_election.5
	title = ASH_election.5.t
	desc = ASH_election.5.d
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes
	
	option = {
		name = ASH_election.5.a
		hidden_effect = {
			set_variable_to_random = {
				var = qasim_bonus
				min = 0
				max = 5
				integer = yes
			}
			set_variable_to_random = {
				var = abdul_bonus
				min = 0
				max = 5
				integer = yes
			}
			set_variable_to_random = {
				var = basir_bonus
				min = 0
				max = 5
				integer = yes
			}
			add_to_variable = {
				var = qasim_vote
				value = qasim_bonus
			}
			add_to_variable = {
				var = abdul_vote
				value = abdul_bonus
			}
			add_to_variable = {
				var = basir_vote
				value = basir_bonus
			}
			clear_variable = qasim_bonus
			clear_variable = abdul_bonus
			clear_variable = basir_bonus
			if = {
				limit = {
					check_variable = {
						var = qasim_vote
						value = abdul_vote
						compare = greater_than
					}
					check_variable = {
						var = qasim_vote
						value = basir_vote
						compare = greater_than
					}
				}
				set_country_flag = qasim_win
			}
			else_if = {
				limit = {
					check_variable = {
						var = abdul_vote
						value = qasim_vote
						compare = greater_than
					}
					check_variable = {
						var = abdul_vote
						value = basir_vote
						compare = greater_than
					}
				}
				set_country_flag = abdul_win
			}
			else_if = {
				limit = {
					check_variable = {
						var = basir_vote
						value = qasim_vote
						compare = greater_than
					}
					check_variable = {
						var = basir_vote
						value = abdul_vote
						compare = greater_than
					}
				}
				set_country_flag = basir_win
			} # if it gets this far there is a tie
			else_if = { # Qasim wins tie with Basir
				limit = {
					check_variable = {
						var = qasim_vote
						value = basir_bonus
						compare = equals
					}
					check_variable = {
						var = qasim_vote
						value = abdul_vote
						compare = greater_than
					}
				}
				set_country_flag = qasim_win
			}
			else = { # Abdul wins all ties
				set_country_flag = abdul_win
			}
			clear_variable = qasim_vote
			clear_variable = abdul_vote
			clear_variable = basir_vote
			
		}
		country_event = {
			id = ASH_election.6 # Results
		}
	}
}
