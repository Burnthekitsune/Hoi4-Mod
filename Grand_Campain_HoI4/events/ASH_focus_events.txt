country_event = { # Invite to faction
	id = ASH_focus.1
	title = ASH_focus.1.t
	desc = ASH_focus.1.d
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes
	
	option = { # Yes
		name = ASH_focus.1.a
		X10 = {
			add_to_faction = ROOT
		}
		add_to_tech_sharing_group = communist_front_tech_group
		if = {
			limit = {
				has_global_flag = UCF
			}
			add_ideas = ASH_unified_communist_front
		}
		hidden_effect = {
			add_popularity = {
				ideology = communism
				popularity = 0.1
			}
			add_ai_strategy = {
				type = alliance
				id = X10
				value = 200
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				NOT = {
					has_government = communism
				}
				factor = 0
			}
		}
	}
	
	option = { # NO
		name = ASH_focus.1.b
		ai_chance = {
			base = 1
			modifier = {
				has_government = communism
				factor = 0
			}
		}
	}
}
